function validateGeneratedQuestion(question: any): ValidationResult {
  const errors: string[] = [];
  
  // Schema validation
  if (!question.id || !question.question || !question.correctAnswer) {
    errors.push("Missing required fields");
  }
  
  // Type validation
  if (question.type !== 'multiple-choice' && question.type !== 'multi-select') {
    errors.push("Invalid question type");
  }
  
  // Options validation
  if (!question.options || question.options.length < 2) {
    errors.push("Insufficient answer options");
  }
  
  // Correct answer validation
  if (Array.isArray(question.correctAnswer)) {
    // Multi-select: check all answers exist in options
    const validOptions = question.options.map(o => o.id);
    const invalidAnswers = question.correctAnswer.filter(a => !validOptions.includes(a));
    if (invalidAnswers.length > 0) {
      errors.push(`Invalid correct answers: ${invalidAnswers.join(', ')}`);
    }
  } else {
    // Single choice: check answer exists
    if (!question.options.find(o => o.id === question.correctAnswer)) {
      errors.push("Correct answer not in options");
    }
  }
  
  // Topic validation
  if (!topicExists(question.topicId, question.subtopicId)) {
    errors.push("Invalid topic reference");
  }
  
  return {
    valid: errors.length === 0,
    errors,
    warnings: [] // Could add soft checks
  };
}

Validator Decision Logic:

interface ValidationResult {
  approved: boolean;
  confidence: number;
  scores: ValidationScores;
  overallScore: number;
  issues: string[];
  suggestions: string[];
  factualConcerns: string[];
}

function makeValidationDecision(result: ValidationResult): ValidationDecision {
  // High confidence, high score → AUTO-APPROVE
  if (result.confidence >= 0.85 && result.overallScore >= 8.0) {
    return {
      status: 'approved',
      requiresReview: false
    };
  }
  
  // Low score → AUTO-REJECT
  if (result.overallScore < 6.0) {
    return {
      status: 'rejected',
      requiresReview: false,
      reason: result.issues.join('; ')
    };
  }
  
  // Medium score or low confidence → FLAG FOR REVIEW
  if (result.confidence < 0.85 || result.overallScore < 8.0) {
    return {
      status: 'flagged',
      requiresReview: true,
      reason: 'Validator uncertain or quality concerns'
    };
  }
  
  // Has factual concerns → FLAG FOR REVIEW
  if (result.factualConcerns.length > 0) {
    return {
      status: 'flagged',
      requiresReview: true,
      reason: 'Factual accuracy needs verification'
    };
  }
}

// File: /lib/ai/validator-agent.ts

interface ValidatorConfig {
  provider: AIProvider;
  model: string;
  approvalThreshold: number;  // Default: 8.0
  confidenceThreshold: number; // Default: 0.85
}

class QuestionValidatorAgent {
  constructor(config: ValidatorConfig) {
    this.config = config;
  }
  
  async validateQuestion(
    question: Question,
    context: ValidationContext
  ): Promise<ValidationResult> {
    // Build validation prompt with question and context
    const prompt = this.buildValidationPrompt(question, context);
    
    // Call AI provider
    const response = await this.config.provider.chat(prompt);
    
    // Parse structured response
    const result = JSON.parse(response);
    
    // Make decision
    return this.makeDecision(result);
  }
  
  async validateBatch(
    questions: Question[]
  ): Promise<BatchValidationResult> {
    // Parallel validation for speed
    const results = await Promise.all(
      questions.map(q => this.validateQuestion(q, context))
    );
    
    return {
      approved: results.filter(r => r.status === 'approved'),
      rejected: results.filter(r => r.status === 'rejected'),
      flagged: results.filter(r => r.status === 'flagged'),
      stats: this.calculateStats(results)
    };
  }
}
